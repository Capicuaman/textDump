# --------------------------------------------------------------------------
# 1. GIT REPOSITORY MAINTENANCE
# --------------------------------------------------------------------------

# Fetch updates from remotes every 2 hours (doesn't merge, just fetches)
0 */2 * * * cd /home/capicuaman/Documents/textDump && git fetch --all >> /home/capicuaman/logs/git_fetch.log 2>&1

# Weekly git garbage collection (optimize repos) - Sunday 2am
0 2 * * 0 cd /home/capicuaman/Documents/textDump && git gc --auto >> /home/capicuaman/logs/git_gc.log 2>&1

# Check for uncommitted changes and send reminder - end of workday
0 20 * * 1-5 cd /home/capicuaman/Documents/textDump && git status --short | grep -q . && echo "You have uncommitted changes in textDump" >> /home/capicuaman/logs/git_reminders.log

# 2. BACKUP AUTOMATION
# --------------------------------------------------------------------------

# Daily backup at 1am (when you're asleep)

# Weekly backup to external location - Sunday 3am
0 3 * * 0 rsync -avz /home/capicuaman/Documents/ /media/external_drive/documents_backup_$(date +\%Y\%m\%d)/ >> /home/capicuaman/logs/backup_weekly.log 2>&1
